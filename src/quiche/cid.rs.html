39/// Used to calculate the cap for the queue of retired connection IDs for which
40/// a RETIRED_CONNECTION_ID frame have not been sent, as a multiple of
41/// `active_conn_id_limit` (see RFC 9000, section 5.1.2).
42const RETIRED_CONN_ID_LIMIT_MULTIPLIER: usize = 3;
43
44#[derive(Default)]
45struct BoundedConnectionIdSeqSet {
46    /// The inner set.
47    inner: HashSet<u64>,
48
49    /// The maximum number of elements that the set can have.
50    capacity: usize,
51}
52
53impl BoundedConnectionIdSeqSet {
54    /// Creates a set bounded by `capacity`.
55    fn new(capacity: usize) -> Self {
56        Self {
57            inner: HashSet::new(),
58            capacity,
59        }
60    }
61
62    fn insert(&mut self, e: u64) -> Result<bool> {
63        if self.inner.len() >= self.capacity {
64            return Err(Error::IdLimit);
65        }
66
67        Ok(self.inner.insert(e))
68    }
69
70    fn remove(&mut self, e: &u64) -> bool {
71        self.inner.remove(e)
72    }
73
74    fn is_empty(&self) -> bool {
75        self.inner.is_empty()
76    }
77}

86    pub seq: u64,
87
88    /// Its associated reset token. Initial CIDs may not have any reset token.

90
91    /// The path identifier using this CID, if any.
92    pub path_id: Option<usize>,
93}

95#[derive(Default)]
96struct BoundedNonEmptyConnectionIdVecDeque {
97    /// The inner `VecDeque`.
98    inner: VecDeque<ConnectionIdEntry>,
99
100    /// The maximum number of elements that the `VecDeque` can have.
101    capacity: usize,
102}
103
104impl BoundedNonEmptyConnectionIdVecDeque {
105    /// Creates a `VecDeque` bounded by `capacity` and inserts
106    /// `initial_entry` in it.
107    fn new(capacity: usize, initial_entry: ConnectionIdEntry) -> Self {
108        let mut inner = VecDeque::with_capacity(1);
109        inner.push_back(initial_entry);
110        Self { inner, capacity }
111    }
112
113    /// Updates the maximum capacity of the inner `VecDeque` to `new_capacity`.
114    /// Does nothing if `new_capacity` is lower or equal to the current
115    /// `capacity`.
116    fn resize(&mut self, new_capacity: usize) {
117        if new_capacity > self.capacity {
118            self.capacity = new_capacity;
119        }
120    }
121
122    /// Returns the oldest inserted entry still present in the `VecDeque`.
123    fn get_oldest(&self) -> &ConnectionIdEntry {
124        self.inner.front().expect("vecdeque is empty")
125    }
126
127    /// Gets a immutable reference to the entry having the provided `seq`.
128    fn get(&self, seq: u64) -> Option<&ConnectionIdEntry> {
129        // We need to iterate over the whole map to find the key.
130        self.inner.iter().find(|e| e.seq == seq)
131    }
132
133    /// Gets a mutable reference to the entry having the provided `seq`.
134    fn get_mut(&mut self, seq: u64) -> Option<&mut ConnectionIdEntry> {
135        // We need to iterate over the whole map to find the key.
136        self.inner.iter_mut().find(|e| e.seq == seq)
137    }
138
139    /// Returns an iterator over the entries in the `VecDeque`.
140    fn iter(&self) -> impl Iterator<Item = &ConnectionIdEntry> {
141        self.inner.iter()
142    }
143
144    /// Returns the number of elements in the `VecDeque`.
145    fn len(&self) -> usize {
146        self.inner.len()
147    }
148
149    /// Inserts the provided entry in the `VecDeque`.
150    ///
151    /// This method ensures the unicity of the `seq` associated to an entry. If
152    /// an entry has the same `seq` than `e`, this method updates the entry in
153    /// the `VecDeque` and the number of stored elements remains unchanged.
154    ///
155    /// If inserting a new element would exceed the collection's capacity, this
156    /// method raises an [`IdLimit`].
157    ///
158    /// [`IdLimit`]: enum.Error.html#IdLimit
159    fn insert(&mut self, e: ConnectionIdEntry) -> Result<()> {
160        // Ensure we don't have duplicates.
161        match self.get_mut(e.seq) {
162            Some(oe) => *oe = e,
163            None => {
164                if self.inner.len() >= self.capacity {
165                    return Err(Error::IdLimit);
166                }
167                self.inner.push_back(e);
168            },
169        };
170        Ok(())
171    }
172
173    /// Removes all the elements in the collection and inserts the provided one.
174    fn clear_and_insert(&mut self, e: ConnectionIdEntry) {
175        self.inner.clear();
176        self.inner.push_back(e);
177    }
178
179    /// Removes the element in the collection having the provided `seq`.
180    ///
181    /// If this method is called when there remains a single element in the
182    /// collection, this method raises an [`OutOfIdentifiers`].
183    ///
184    /// Returns `Some` if the element was in the collection and removed, or
185    /// `None` if it was not and nothing was modified.
186    ///
187    /// [`OutOfIdentifiers`]: enum.Error.html#OutOfIdentifiers
188    fn remove(&mut self, seq: u64) -> Result<Option<ConnectionIdEntry>> {
189        if self.inner.len() <= 1 {
190            return Err(Error::OutOfIdentifiers);
191        }
192
193        Ok(self
194            .inner
195            .iter()
196            .position(|e| e.seq == seq)
197            .and_then(|index| self.inner.remove(index)))
198    }
199}

202pub struct ConnectionIdentifiers {
203    /// All the Destination Connection IDs provided by our peer.

242    /// Creates a new `ConnectionIdentifiers` with the specified destination
243    /// connection ID limit and initial source Connection ID. The destination
244    /// Connection ID is set to the empty one.
245    pub fn new(
246        mut destination_conn_id_limit: usize, initial_scid: &ConnectionId,
247        initial_path_id: usize, reset_token: Option<u128>,
248    ) -> ConnectionIdentifiers {
249        // It must be at least 2.
250        if destination_conn_id_limit < 2 {
251            destination_conn_id_limit = 2;
252        }
253
254        // Initially, the limit of active source connection IDs is 2.
255        let source_conn_id_limit = 2;
256
257        // Record the zero-length SCID status.
258        let zero_length_scid = initial_scid.is_empty();
259
260        let initial_scid =
261            ConnectionId::from_ref(initial_scid.as_ref()).into_owned();
262
263        // We need to track up to (2 * source_conn_id_limit - 1) source
264        // Connection IDs when the host wants to force their renewal.
265        let scids = BoundedNonEmptyConnectionIdVecDeque::new(
266            2 * source_conn_id_limit - 1,
267            ConnectionIdEntry {
268                cid: initial_scid,
269                seq: 0,
270                reset_token,
271                path_id: Some(initial_path_id),
272            },
273        );
274
275        let dcids = BoundedNonEmptyConnectionIdVecDeque::new(
276            destination_conn_id_limit,
277            ConnectionIdEntry {
278                cid: ConnectionId::default(),
279                seq: 0,
280                reset_token: None,
281                path_id: Some(initial_path_id),
282            },
283        );
284
285        // Because we already inserted the initial SCID.
286        let next_scid_seq = 1;
287        ConnectionIdentifiers {
288            scids,
289            dcids,
290            retire_dcid_seqs: BoundedConnectionIdSeqSet::new(
291                destination_conn_id_limit * RETIRED_CONN_ID_LIMIT_MULTIPLIER,
292            ),
293            next_scid_seq,
294            source_conn_id_limit,
295            zero_length_scid,
296            ..Default::default()
297        }
298    }

300    /// Sets the maximum number of source connection IDs our peer allows us.
301    pub fn set_source_conn_id_limit(&mut self, v: u64) {
302        // Bound conn id limit so our scids queue sizing is valid.
303        let v = std::cmp::min(v, (usize::MAX / 2) as u64) as usize;
304
305        // It must be at least 2.
306        if v >= 2 {
307            self.source_conn_id_limit = v;
308            // We need to track up to (2 * source_conn_id_limit - 1) source
309            // Connection IDs when the host wants to force their renewal.
310            self.scids.resize(2 * v - 1);
311        }
312    }

328    /// Adds a new source identifier, and indicates whether it should be
329    /// advertised through a `NEW_CONNECTION_ID` frame or not.
330    ///
331    /// At any time, the peer cannot have more Destination Connection IDs than
332    /// the maximum number of active Connection IDs it negotiated. In such case
333    /// (i.e., when [`active_source_cids()`] - `peer_active_conn_id_limit` = 0,
334    /// if the caller agrees to request the removal of previous connection IDs,
335    /// it sets the `retire_if_needed` parameter. Otherwise, an [`IdLimit`] is
336    /// returned.
337    ///
338    /// Note that setting `retire_if_needed` does not prevent this function from
339    /// returning an [`IdLimit`] in the case the caller wants to retire still
340    /// unannounced Connection IDs.
341    ///
342    /// When setting the initial Source Connection ID, the `reset_token` may be
343    /// `None`. However, other Source CIDs must have an associated
344    /// `reset_token`. Providing `None` as the `reset_token` for non-initial
345    /// SCIDs raises an [`InvalidState`].
346    ///
347    /// In the case the provided `cid` is already present, it does not add it.
348    /// If the provided `reset_token` differs from the one already registered,
349    /// returns an `InvalidState`.
350    ///
351    /// Returns the sequence number associated to that new source identifier.
352    ///
353    /// [`active_source_cids()`]:  struct.ConnectionIdentifiers.html#method.active_source_cids
354    /// [`InvalidState`]: enum.Error.html#InvalidState
355    /// [`IdLimit`]: enum.Error.html#IdLimit
356    pub fn new_scid(
357        &mut self, cid: ConnectionId<'static>, reset_token: Option<u128>,
358        advertise: bool, path_id: Option<usize>, retire_if_needed: bool,
359    ) -> Result<u64> {
360        if self.zero_length_scid {
361            return Err(Error::InvalidState);
362        }
363
364        // Check whether the number of source Connection IDs does not exceed the
365        // limit. If the host agrees to retire old CIDs, it can store up to
366        // (2 * source_active_conn_id - 1) source CIDs. This limit is enforced
367        // when calling `self.scids.insert()`.
368        if self.scids.len() >= self.source_conn_id_limit {
369            if !retire_if_needed {
370                return Err(Error::IdLimit);
371            }
372
373            // We need to retire the lowest one.
374            self.retire_prior_to = self.lowest_usable_scid_seq()? + 1;
375        }
376
377        let seq = self.next_scid_seq;
378
379        if reset_token.is_none() && seq != 0 {
380            return Err(Error::InvalidState);
381        }
382
383        // Check first that the SCID has not been inserted before.
384        if let Some(e) = self.scids.iter().find(|e| e.cid == cid) {
385            if e.reset_token != reset_token {
386                return Err(Error::InvalidState);
387            }
388            return Ok(e.seq);
389        }
390
391        self.scids.insert(ConnectionIdEntry {
392            cid,
393            seq,
394            reset_token,
395            path_id,
396        })?;
397        self.next_scid_seq += 1;
398
399        self.mark_advertise_new_scid_seq(seq, advertise);
400
401        Ok(seq)
402    }

419    /// Adds a new Destination Connection ID (originating from a
420    /// NEW_CONNECTION_ID frame) and process all its related metadata.
421    ///
422    /// Returns an error if the provided Connection ID or its metadata are
423    /// invalid.
424    ///
425    /// Returns a list of tuples (DCID sequence number, Path ID), containing the
426    /// sequence number of retired DCIDs that were linked to their respective
427    /// Path ID.
428    pub fn new_dcid(
429        &mut self, cid: ConnectionId<'static>, seq: u64, reset_token: u128,
430        retire_prior_to: u64, retired_path_ids: &mut SmallVec<[(u64, usize); 1]>,
431    ) -> Result<()> {
432        if self.zero_length_dcid {
433            return Err(Error::InvalidState);
434        }
435
436        // If an endpoint receives a NEW_CONNECTION_ID frame that repeats a
437        // previously issued connection ID with a different Stateless Reset
438        // Token field value or a different Sequence Number field value, or if a
439        // sequence number is used for different connection IDs, the endpoint
440        // MAY treat that receipt as a connection error of type
441        // PROTOCOL_VIOLATION.
442        if let Some(e) = self.dcids.iter().find(|e| e.cid == cid || e.seq == seq)
443        {
444            if e.cid != cid || e.seq != seq || e.reset_token != Some(reset_token)
445            {
446                return Err(Error::InvalidFrame);
447            }
448            // The identifier is already there, nothing to do.
449            return Ok(());
450        }
451
452        // The value in the Retire Prior To field MUST be less than or equal to
453        // the value in the Sequence Number field. Receiving a value in the
454        // Retire Prior To field that is greater than that in the Sequence
455        // Number field MUST be treated as a connection error of type
456        // FRAME_ENCODING_ERROR.
457        if retire_prior_to > seq {
458            return Err(Error::InvalidFrame);
459        }
460
461        // An endpoint that receives a NEW_CONNECTION_ID frame with a sequence
462        // number smaller than the Retire Prior To field of a previously
463        // received NEW_CONNECTION_ID frame MUST send a corresponding
464        // RETIRE_CONNECTION_ID frame that retires the newly received connection
465        // ID, unless it has already done so for that sequence number.
466        if seq < self.largest_peer_retire_prior_to {
467            self.mark_retire_dcid_seq(seq, true)?;
468            return Ok(());
469        }
470
471        if seq > self.largest_destination_seq {
472            self.largest_destination_seq = seq;
473        }
474
475        let new_entry = ConnectionIdEntry {
476            cid: cid.clone(),
477            seq,
478            reset_token: Some(reset_token),
479            path_id: None,
480        };
481
482        let mut retired_dcid_queue_err = None;
483
484        // A receiver MUST ignore any Retire Prior To fields that do not
485        // increase the largest received Retire Prior To value.
486        //
487        // After processing a NEW_CONNECTION_ID frame and adding and retiring
488        // active connection IDs, if the number of active connection IDs exceeds
489        // the value advertised in its active_connection_id_limit transport
490        // parameter, an endpoint MUST close the connection with an error of type
491        // CONNECTION_ID_LIMIT_ERROR.
492        if retire_prior_to > self.largest_peer_retire_prior_to {
493            let retired = &mut self.retire_dcid_seqs;
494
495            // The insert entry MUST have a sequence higher or equal to the ones
496            // being retired.
497            if new_entry.seq < retire_prior_to {
498                return Err(Error::OutOfIdentifiers);
499            }
500
501            // To avoid exceeding the capacity of the inner `VecDeque`, we first
502            // remove the elements and then insert the new one.
503            let index = self
504                .dcids
505                .inner
506                .partition_point(|e| e.seq < retire_prior_to);
507
508            for e in self.dcids.inner.drain(..index) {
509                if let Some(pid) = e.path_id {
510                    retired_path_ids.push((e.seq, pid));
511                }
512
513                if let Err(e) = retired.insert(e.seq) {
514                    // Delay propagating the error as we need to try to insert
515                    // the new DCID first.
516                    retired_dcid_queue_err = Some(e);
517                    break;
518                }
519            }
520
521            self.largest_peer_retire_prior_to = retire_prior_to;
522        }
523
524        // Note that if no element has been retired and the `VecDeque` reaches
525        // its capacity limit, this will raise an `IdLimit`.
526        self.dcids.insert(new_entry)?;
527
528        // Propagate the error triggered when inserting a retired DCID seq to
529        // the queue.
530        if let Some(e) = retired_dcid_queue_err {
531            return Err(e);
532        }
533
534        Ok(())
535    }

537    /// Retires the Source Connection ID having the provided sequence number.
538    ///
539    /// In case the retired Connection ID is the same as the one used by the
540    /// packet requesting the retiring, or if the retired sequence number is
541    /// greater than any previously advertised sequence numbers, it returns an
542    /// [`InvalidState`].
543    ///
544    /// Returns the path ID that was associated to the retired CID, if any.
545    ///
546    /// [`InvalidState`]: enum.Error.html#InvalidState
547    pub fn retire_scid(
548        &mut self, seq: u64, pkt_dcid: &ConnectionId,
549    ) -> Result<Option<usize>> {
550        if seq >= self.next_scid_seq {
551            return Err(Error::InvalidState);
552        }
553
554        let pid = if let Some(e) = self.scids.remove(seq)? {
555            if e.cid == *pkt_dcid {
556                return Err(Error::InvalidState);
557            }
558
559            // Notifies the application.
560            self.retired_scids.push_back(e.cid);
561
562            // Retiring this SCID may increase the retire prior to.
563            let lowest_scid_seq = self.lowest_usable_scid_seq()?;
564            self.retire_prior_to = lowest_scid_seq;
565
566            e.path_id
567        } else {
568            None
569        };
570
571        Ok(pid)
572    }

574    /// Retires the Destination Connection ID having the provided sequence
575    /// number.
576    ///
577    /// If the caller tries to retire the last destination Connection ID, this
578    /// method triggers an [`OutOfIdentifiers`].
579    ///
580    /// If the caller tries to retire a non-existing Destination Connection
581    /// ID sequence number, this method returns an [`InvalidState`].
582    ///
583    /// Returns the path ID that was associated to the retired CID, if any.
584    ///
585    /// [`OutOfIdentifiers`]: enum.Error.html#OutOfIdentifiers
586    /// [`InvalidState`]: enum.Error.html#InvalidState
587    pub fn retire_dcid(&mut self, seq: u64) -> Result<Option<usize>> {
588        if self.zero_length_dcid {
589            return Err(Error::InvalidState);
590        }
591
592        let e = self.dcids.remove(seq)?.ok_or(Error::InvalidState)?;
593
594        self.mark_retire_dcid_seq(seq, true)?;
595
596        Ok(e.path_id)
597    }

624    /// Gets the minimum Source Connection ID sequence number whose removal has
625    /// not been requested yet.
626    #[inline]
627    pub fn lowest_usable_scid_seq(&self) -> Result<u64> {
628        self.scids
629            .iter()
630            .filter_map(|e| {
631                if e.seq >= self.retire_prior_to {
632                    Some(e.seq)
633                } else {
634                    None
635                }
636            })
637            .min()
638            .ok_or(Error::InvalidState)
639    }

677    #[inline]
678    pub fn available_scids(&self) -> usize {
679        self.scids.iter().filter(|e| e.path_id.is_none()).count()
680    }
681
682    /// Returns the number of Destination Connection IDs that have not been
683    /// assigned to a path yet.
684    ///
685    /// Note that this function returns 0 if the host uses zero length
686    /// Destination Connection IDs.
687    #[inline]
688    pub fn available_dcids(&self) -> usize {
689        if self.zero_length_dcid() {
690            return 0;
691        }
692        self.dcids.iter().filter(|e| e.path_id.is_none()).count()
693    }

695    /// Returns the oldest active source Connection ID of this connection.
696    #[inline]
697    pub fn oldest_scid(&self) -> &ConnectionIdEntry {
698        self.scids.get_oldest()
699    }
700
701    /// Returns the oldest known active destination Connection ID of this
702    /// connection.
703    ///
704    /// Note that due to e.g., reordering at reception side, the oldest known
705    /// active destination Connection ID is not necessarily the one having the
706    /// lowest sequence.
707    #[inline]
708    pub fn oldest_dcid(&self) -> &ConnectionIdEntry {
709        self.dcids.get_oldest()
710    }

712    /// Adds or remove the source Connection ID sequence number from the
713    /// source Connection ID set that need to be advertised to the peer through
714    /// NEW_CONNECTION_ID frames.
715    #[inline]
716    pub fn mark_advertise_new_scid_seq(
717        &mut self, scid_seq: u64, advertise: bool,
718    ) {
719        if advertise {
720            self.advertise_new_scid_seqs.push_back(scid_seq);
721        } else if let Some(index) = self
722            .advertise_new_scid_seqs
723            .iter()
724            .position(|s| *s == scid_seq)
725        {
726            self.advertise_new_scid_seqs.remove(index);
727        }
728    }
729
730    /// Adds or remove the destination Connection ID sequence number from the
731    /// retired destination Connection ID set that need to be advertised to the
732    /// peer through RETIRE_CONNECTION_ID frames.
733    #[inline]
734    pub fn mark_retire_dcid_seq(
735        &mut self, dcid_seq: u64, retire: bool,
736    ) -> Result<()> {
737        if retire {
738            self.retire_dcid_seqs.insert(dcid_seq)?;
739        } else {
740            self.retire_dcid_seqs.remove(&dcid_seq);
741        }
742
743        Ok(())
744    }
745
746    /// Gets a source Connection ID's sequence number requiring advertising it
747    /// to the peer through NEW_CONNECTION_ID frame, if any.
748    ///
749    /// If `Some`, it always returns the same value until it has been removed
750    /// using `mark_advertise_new_scid_seq`.
751    #[inline]
752    pub fn next_advertise_new_scid_seq(&self) -> Option<u64> {
753        self.advertise_new_scid_seqs.front().copied()
754    }
755
756    /// Returns a copy of the set of destination Connection IDs's sequence
757    /// numbers to send RETIRE_CONNECTION_ID frames.
758    ///
759    /// Note that the set includes sequence numbers at the time the copy was
760    /// created. To account for newly inserted or removed sequence numbers, a
761    /// new copy needs to be created.
762    #[inline]
763    pub fn retire_dcid_seqs(&self) -> HashSet<u64> {
764        self.retire_dcid_seqs.inner.clone()
765    }
766
767    /// Returns true if there are new source Connection IDs to advertise.
768    #[inline]
769    pub fn has_new_scids(&self) -> bool {
770        !self.advertise_new_scid_seqs.is_empty()
771    }
772
773    /// Returns true if there are retired destination Connection IDs to\
774    /// advertise.
775    #[inline]
776    pub fn has_retire_dcids(&self) -> bool {
777        !self.retire_dcid_seqs.is_empty()
778    }
779
780    /// Returns whether zero-length source CIDs are used.
781    #[inline]
782    pub fn zero_length_scid(&self) -> bool {
783        self.zero_length_scid
784    }
785
786    /// Returns whether zero-length destination CIDs are used.
787    #[inline]
788    pub fn zero_length_dcid(&self) -> bool {
789        self.zero_length_dcid
790    }
791
792    /// Gets the NEW_CONNECTION_ID frame related to the source connection ID
793    /// with sequence `seq_num`.
794    pub fn get_new_connection_id_frame_for(
795        &self, seq_num: u64,
796    ) -> Result<frame::Frame> {
797        let e = self.scids.get(seq_num).ok_or(Error::InvalidState)?;
798        Ok(frame::Frame::NewConnectionId {
799            seq_num,
800            retire_prior_to: self.retire_prior_to,
801            conn_id: e.cid.to_vec(),
802            reset_token: e.reset_token.ok_or(Error::InvalidState)?.to_be_bytes(),
803        })
804    }
805
806    /// Returns the number of source Connection IDs that are active. This is
807    /// only meaningful if the host uses non-zero length Source Connection IDs.
808    #[inline]
809    pub fn active_source_cids(&self) -> usize {
810        self.scids.len()
811    }
812
813    /// Returns the number of source Connection IDs that are retired. This is
814    /// only meaningful if the host uses non-zero length Source Connection IDs.
815    #[inline]
816    pub fn retired_source_cids(&self) -> usize {
817        self.retired_scids.len()
818    }
819
820    pub fn pop_retired_scid(&mut self) -> Option<ConnectionId<'static>> {
821        self.retired_scids.pop_front()
822    }

890    fn new_dcid_event() {
891        let (scid, _) = create_cid_and_reset_token(16);
892        let (dcid, _) = create_cid_and_reset_token(16);
893
894        let mut retired_path_ids = SmallVec::new();
895
896        let mut ids = ConnectionIdentifiers::new(2, &scid, 0, None);
897        ids.set_initial_dcid(dcid, None, Some(0));
898
899        assert_eq!(ids.available_dcids(), 0);
900        assert_eq!(ids.dcids.len(), 1);
901
902        let (dcid2, rt2) = create_cid_and_reset_token(16);
903
904        assert_eq!(
905            ids.new_dcid(dcid2, 1, rt2, 0, &mut retired_path_ids),
906            Ok(()),
907        );
908        assert_eq!(retired_path_ids, SmallVec::from_buf([]));
909        assert_eq!(ids.available_dcids(), 1);
910        assert_eq!(ids.dcids.len(), 2);
911
912        // Now we assume that the client wants to advertise more source
913        // Connection IDs than the advertised limit. This is valid if it
914        // requests its peer to retire enough Connection IDs to fit within the
915        // limits.
916        let (dcid3, rt3) = create_cid_and_reset_token(16);
917        assert_eq!(
918            ids.new_dcid(dcid3, 2, rt3, 1, &mut retired_path_ids),
919            Ok(())
920        );
921        assert_eq!(retired_path_ids, SmallVec::from_buf([(0, 0)]));
922        // The CID module does not handle path replacing. Fake it now.
923        ids.link_dcid_to_path_id(1, 0).unwrap();
924        assert_eq!(ids.available_dcids(), 1);
925        assert_eq!(ids.dcids.len(), 2);
926        assert!(ids.has_retire_dcids());
927        assert_eq!(ids.retire_dcid_seqs().iter().next(), Some(&0));
928
929        // Fake RETIRE_CONNECTION_ID sending.
930        let _ = ids.mark_retire_dcid_seq(0, false);
931        assert!(!ids.has_retire_dcids());
932        assert_eq!(ids.retire_dcid_seqs().iter().next(), None);
933
934        // Now tries to experience CID retirement. If the server tries to remove
935        // non-existing DCIDs, it fails.
936        assert_eq!(ids.retire_dcid(0), Err(Error::InvalidState));
937        assert_eq!(ids.retire_dcid(3), Err(Error::InvalidState));
938        assert!(!ids.has_retire_dcids());
939        assert_eq!(ids.dcids.len(), 2);
940
941        // Now it removes DCID with sequence 1.
942        assert_eq!(ids.retire_dcid(1), Ok(Some(0)));
943        // The CID module does not handle path replacing. Fake it now.
944        ids.link_dcid_to_path_id(2, 0).unwrap();
945        assert_eq!(ids.available_dcids(), 0);
946        assert!(ids.has_retire_dcids());
947        assert_eq!(ids.retire_dcid_seqs().iter().next(), Some(&1));
948        assert_eq!(ids.dcids.len(), 1);
949
950        // Fake RETIRE_CONNECTION_ID sending.
951        let _ = ids.mark_retire_dcid_seq(1, false);
952        assert!(!ids.has_retire_dcids());
953        assert_eq!(ids.retire_dcid_seqs().iter().next(), None);
954
955        // Trying to remove the last DCID triggers an error.
956        assert_eq!(ids.retire_dcid(2), Err(Error::OutOfIdentifiers));
957        assert_eq!(ids.available_dcids(), 0);
958        assert!(!ids.has_retire_dcids());
959        assert_eq!(ids.dcids.len(), 1);
960    }

971

973        assert_eq!(ids.dcids.len(), 1);
974
975        // Skip DCID #1 (e.g due to packet loss) and insert DCID #2.

981        assert!(ids.new_dcid(dcid, 3, rt, 2, &mut retired_path_ids).is_ok());
982        assert_eq!(ids.dcids.len(), 2);
983
984        let (dcid, rt) = create_cid_and_reset_token(16);
985        assert!(ids.new_dcid(dcid, 4, rt, 3, &mut retired_path_ids).is_ok());
986        assert_eq!(ids.dcids.len(), 2);
987
988        // Insert DCID #1 (e.g due to packet reordering).
989        let (dcid, rt) = create_cid_and_reset_token(16);
990        assert!(ids.new_dcid(dcid, 1, rt, 0, &mut retired_path_ids).is_ok());
991        assert_eq!(ids.dcids.len(), 2);
992
993        // Try inserting DCID #1 again (e.g. due to retransmission).
994        let (dcid, rt) = create_cid_and_reset_token(16);
